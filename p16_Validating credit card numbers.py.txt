import re

n = int(input())

for _ in range(n):
    card = input().strip()
    
    # Check basic format
    pattern = r'^[456]\d{3}(-?\d{4}){3}$'
    
    if re.match(pattern, card):
        # Remove hyphens to check consecutive digits
        num = card.replace('-', '')
        
        if re.search(r'(\d)\1\1\1', num):
            print("Invalid")
        else:
            print("Valid")
    else:
        print("Invalid")
